// Toggle Sneak attack
// Requires Dynamic Effects and (optionally) MacroMarker
DynamicEffects.togglePassive("Sneak Attack", "feat");

// Mark tokens with mystery man
const icon = "icons/svg/mystery-man.svg";
let tokens = canvas.tokens.controlled;
let self = this; // otherwise, "this" is not found in the forEach loop.
//console.log(tokens.length);
tokens.forEach( function(token) {
 token.toggleEffect(icon);
 MacroMarker.toggle(self, { entity: token });
});
