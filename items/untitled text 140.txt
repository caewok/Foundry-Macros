// cmpFn: if a > b return true or a positive value
function binarySearch(arr, element, cmpFn) {
  if ( !cmpFn(element, arr[0]) ) return 0;

  let n = arr.length;
  if ( cmpFn(element, arr[n - 1]) ) return n;

  let m = 0;
  while ( m <= n ) {
    const k = ( n + m ) >> 1;
    const cmp = cmpFn(element, arr[k]);
    if ( cmp > 0 ) m = k + 1;
    else if ( cmp < 0 ) n = k - 1;
    else return k;
  }
  return -m - 1;
}

contains(t,i) {
  let s=!1;
  const r=this.points.length/2;
  for(let n=0,o=r-1;n<r;o=n++) {
    const a=this.points[n*2],h=this.points[n*2+1],l=this.points[o*2],c=this.points[o*2+1];
    h>i!=c>i&&t<(l-a)*((i-h)/(c-h))+a&&(s=!s)
  }
  return s
}


function contains(poly, x, y) {
  let s = !1;
  const r = poly.points.length / 2;
  for ( let n = 0, o = r - 1; n < r; o = n++ ) {
    const a = poly.points[n * 2];
    const h = poly.points[(n * 2) + 1];
    const l = poly.points[o * 2];
    const c = poly.points[(o * 2) + 1];
    h > y != c > y && x < (l - a) * ((y - h) / (c - h)) + a && (s = !s)
  }
  return s;
}

function multipleContains(poly, testPoints) {
  // Modification of PIXI.Polygon.prototype.contains
  const nPoints = testPoints.length;
  if ( nPoints < 2 ) return undefined;
  const res = new Uint8Array(nPoints * 0.5); // If we really need speed, could use bit packing
  const r = poly.points.length / 2;
  for ( let n = 0, o = r - 1; n < r; o = n++ ) {
    const a = poly.points[n * 2];
    const h = poly.points[(n * 2) + 1];
    const l = poly.points[o * 2];
    const c = poly.points[(o * 2) + 1];

    for ( let i = 0, j = 0; i < nPoints; i += 2, j += 1 ) {
      const x = testPoints[i];
      const y = testPoints[i + 1];
      h > y != c > y && x < (l - a) * ((y - h) / (c - h)) + a && (res[j] = !res[j])
    }
  }
  return res;
}
